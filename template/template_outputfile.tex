% Title of of a Markdown File
% #1 ... depth, as an increasing int from 0
% #2 ... actual title, to be printed visibly
\newcommand{\mezdoctitle}[2]{%
    \ifcase#1
        \addcontentsline{toc}{section}{#2}%
    \or
        \addcontentsline{toc}{subsection}{#2}%
    \or
        \addcontentsline{toc}{subsubsection}{#2}%
    \or
        \addcontentsline{toc}{paragraph}{#2}%
    \else
        \addcontentsline{toc}{subparagraph}{#2}%
    \fi
    \vspace{1em}
    \par\noindent
    \colorbox{zettlrdocument}{%
        \makebox[\dimexpr\linewidth-10pt\relax][l]{%
            \textbf{#2}%
        }%
    }%
    \par\vspace{1em}
}
